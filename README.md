<!-- Add JavaScript for interactive elements -->
<script>
  // Simple mouse follower effect
  document.addEventListener('mousemove', function(e) {
    const badges = document.querySelectorAll('.badge');
    badges.forEach(badge => {
      const x = e.pageX;
      const y = e.pageY;
      const rect = badge.getBoundingClientRect();
      const badgeX = rect.left + rect.width / 2;
      const badgeY = rect.top + rect.height / 2;
      const distance = Math.sqrt(Math.pow(x - badgeX, 2) + Math.pow(y - badgeY, 2));
      if (distance < 100) {
        badge.style.transform = `scale(${1.2 - distance/200})`;
      }
    });
  });
</script>
